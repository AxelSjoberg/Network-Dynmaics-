# Getting the transition matrix to a np matrix data type
Lambda = np.matrix(Lambda)

# Calculating w and w_max (sometimes called w_star in lectures)
w = Lambda.dot(np.ones((len(Lambda), 1)))
w = [item for sublist in np.asarray(w).tolist() for item in sublist]
w_max = np.max(w)

# Normalized transition matrix
P_norm = normalize(Lambda, norm = 'l1')

# Calculating the P_bar matrix 
P_bar_ij = Lambda / w_max
P_bar_ii = np.diagflat(np.ones((len(Lambda), 1)) - P_bar_ij.dot(np.ones((len(Lambda), 1))))
P_bar = P_bar_ii + P_bar_ij
P_bar = np.asarray(P_bar)

# Calculates the theoretical hit time from all the nodes to the target_set S
def hit_time_theoretical(P,S,w):
    P_hat = np.delete(P, S, 0) #Remove the rows correspoding to the target set 
    P_hat = np.delete(P_hat, S, 1) #Remove the columns correspoding to the target set 
    w_r  = np.delete(w,S) # Creating a list containg the degree of all the nodes but those in our set S.
    
    # Calculate the hit time for the nodes not in set S.
    w_r = [1/i for i in w_r]
    x = np.linalg.solve(np.eye(len(P_hat))-P_hat,np.diagflat(w_r).dot(np.ones((len(w_r),1))))
    
    # This part is only so we get the return time 0 for those node in S and mapping the return time in x to their 
    # corresponding node index. 
    Ts = []
    index = [i for i in range(len(P)) if i not in S]
    flat_list = [item for sublist in x for item in sublist]
    for i in range(0,len(P)):
        if i not in S:
            Ts.append(flat_list.pop(0))
        else:
            Ts.append(0)
    return Ts

# Simulates a continuous time random walk with nbr_of_jumps, prop transition matrix P and with start at node0. 
# node0 is the index in the correspoding P-matrix, not the name of the node. 
def random_walk(nbr_jumps, P, node0):
    nodes = np.arange(0,len(P))
    pos = np.zeros(nbr_jumps) #an array containing the position the particle takes at jump.  
    pos[0] = node0
    t = np.zeros(nbr_jumps) # #an array coontaining the times the poisson clock ticks.
    tnext = -np.log(rand.random())/w_max
    
    for i in range(1,nbr_jumps):
        pos[i] = rand.choices(nodes,P[int(pos[i-1])])[0]
        t[i] = t[i-1] + tnext
        tnext = -np.log(rand.random())/w_max
    
    # Creating an array containing how time between each time tick. 
    tnew = np.concatenate((t,t[-1] + tnext),axis =None)
    tnew = np.diff(tnew)  
    
    # Summing up the time spent in each node and dividing that by the total time. 
    piest = np.zeros(len(P))
    for i in range(0,len(P)):
        ind = np.where(pos == i)
        piest[i] = sum(tnew[ind])/(t[-1]+tnext)
    
    return piest

# Rather straight forward method, calculates the return time using niter iterations and with prop transition matrix P.
def return_time_simulation(niter, P):
    nodes = np.arange(0,len(P))
    Tr = []
    for n in nodes:
        t = 0
        for i in range(0,niter):
            pos = rand.choices(nodes,P[n])[0]
            tnext = -np.log(rand.random())/ w_max
            
            while pos == n:
                pos = rand.choices(nodes,P[int(pos)])[0]
                t = t + tnext
                tnext = -np.log(rand.random())/ w_max #w[pos[-1]]
            
            while pos != n:
                pos = rand.choices(nodes,P[int(pos)])[0]
                t = t + tnext
                tnext = -np.log(rand.random())/ w_max #w[pos[-1]]
                
            t = t + tnext
        Tr_loc = t / niter
        Tr.append(Tr_loc)
    return Tr

def hit_time_simulation(niter, P, S):
    nodes = np.arange(0,len(P))
    node_hits = np.delete(nodes,S)
    x = []
    Ts = []
    for n in node_hits:
        t = 0
        for i in range(0,niter):
            pos = []
            pos.append(rand.choices(nodes,P[n])[0])
            tnext = -np.log(rand.random())/w_max #w[n]
            while pos[-1] not in S:
                pos.append(rand.choices(nodes,P[pos[-1]])[0])
                t = t + tnext
                tnext = -np.log(rand.random())/w_max #w[pos[-1]]
            t = t + tnext
        Ts_loc = t / niter
        x.append(Ts_loc)
    index = [i for i in range(len(P)) if i not in S]
    for i in range(0,len(P)):
        if i not in S:
            Ts.append(x.pop(0))
        else:
            Ts.append(0)
    return Ts

# -----Simulation taking the in-paramters----:
# number of iterations: niter
# transition matrix: P
# Start node: n
# ------- Returns -------
# The average time for a particle to return to node n. 

def return_time_100_particles_simulation(niter, P, n):
    nodes = np.arange(0,len(P))
    
    # Creating the dict pos: all of the particles are the keys and the value is a list containing the positions 
    # the particle have during the simulation. All the particles start in node n. 
    k = list(range(0,100))
    v = []
    for i in range(0,100):
        v.append([n])
    pos = dict(zip(k, v))
    
    # Creating the dict Tr: all of the particles are the keys and the value is a list containing the 
    # time when the node once reach hit node n. 
    k = list(range(0,100))
    v = []
    for i in range(0,100):
        v.append([])
    Tr = dict(zip(k, v))
    
    t = np.zeros(niter) # The number of ticks of the common poisson clock
    tnext = -np.log(rand.random())/100 # Time until the common poisson clock ticks
    
    for i in range(1,niter):
        random_particle = int(100*rand.random()) # Chooses one of the random particles to move
        particle_move_to = rand.choices(nodes,P[int(pos[random_particle][-1])])[0] #Choose where the random particle decides to go
        pos[random_particle].append(particle_move_to) # Note the position where the particle decides to go in pos.
        t[i] = t[i-1] + tnext # update the time
        tnext = -np.log(rand.random())/100 # How long it takes until the common poisson clock ticks again. 
        
        # If we moved the particle back to node n, we take note of at which time this happened.
        if particle_move_to == n:
            Tr[random_particle].append(t[i])
    
    # Creates a new dict compossed of every return time for each particle -> stored in diff. 
    diff = Tr.copy()
    a = dict([(k, np.diff(v).tolist()) for (k, v) in Tr.items()])
    for k in a.keys():
        if len(a[k]) >= 1:
            diff[k] = [diff[k][0]] + a[k]
            
    # Puts all of the return times for all the particles in one list. 
    return_times = [item for sublist in list(diff.values()) for item in sublist]
    
    # Returning sum of all return times, divided by the total number of returns.
    if len(return_times) == 0:
        print('Error need more iterations or try again')
    else:
        return sum(return_times)/len(return_times)

# -----Simmulation based on in-parameters----:
# transition matrix: P
# start node: node0
# max time: max_time
# number of particles: nbr_particles
# Simulates how the number of particles in the different nodes change with time. 
# -------- Returns ---------
# the number of particles in the different nodes at each time step: x
# time vector counting the time when the poisson clock ticked: t
# d - numer of particles in nodes at max_time

def node_perspective_simulation(P, node0, max_time=10, nbr_particles=100):
    n = nbr_particles
    nodes = np.arange(0,len(P))
    v = [0] *len(nodes)
    d = dict(zip(nodes, v))
    d[0] = n
    x = []
    t = [0]
    tnext = -np.log(rand.random())/n
    x.append(list(d.values()))
    
    while t[-1] < max_time:
        random_node = rand.choices(nodes, list(d.values()))[0]
        a = rand.choices(nodes,P[random_node])[0]
        d[random_node] = d[random_node] - 1
        d[a] = d[a] + 1
        t.append(t[-1] + tnext)
        tnext = -np.log(rand.random())/100
        x.append(list(d.values()))
    
    return x,t,d
